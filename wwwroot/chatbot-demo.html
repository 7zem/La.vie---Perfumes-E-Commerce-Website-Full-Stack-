Easy Integration

Add the chatbot to any webpage with just a few lines of code:

<!-- Add this to your HTML head or body -->
<script src="chatbot-widget.js"></script>

<!-- Initialize the chatbot -->
<script>
    new PerfumeChatbotWidget({
        apiUrl: 'http://localhost:5036/api/chatbot',
        position: 'bottom-right',
        title: 'Perfume Assistant',
        subtitle: 'Ask me anything!'
    });
</script>
